Вивести вектор з булевими значеннями TRUE та FALSE. TRUE, якщо в стовпці є елементи >2, FALSE – якщо немає.

bigger_than_too <- function(x) {
    return (sum(x > 2) > 0)
  }
  
  apply(mat, 2, bigger_than_too)
[1] FALSE FALSE FALSE FALSE FALSE
